static inline bool is_vega20(sift_chipid_t id)
{
  switch (id) {
    case SIFT_CHIPID_VEGA20_INSTINCT:
      return true;
    case SIFT_CHIPID_VEGA20_MI50:
      return true;
    case SIFT_CHIPID_VEGA20:
